// ── CALENDAR ──────────────────────────────────────────────────────────────────

function renderCal(){

  document.getElementById('cal-label').textContent=`${MONTHS[calMonth]} ${calYear}`;

  const allEvents=calFilter==='all'?DB.events():DB.events(calFilter);

  const first=new Date(calYear,calMonth,1);

  let dow=first.getDay()-1; if(dow<0)dow=6;

  const dim=new Date(calYear,calMonth+1,0).getDate();

  const pmd=new Date(calYear,calMonth,0).getDate();

  const today=new Date().toISOString().split('T')[0];

  let cells=''; const total=Math.ceil((dow+dim)/7)*7;

  for(let i=0;i<total;i++){

    let day,ds,other=false;

    if(i<dow){ day=pmd-dow+i+1; const m2=calMonth===0?11:calMonth-1,y2=calMonth===0?calYear-1:calYear; ds=`${y2}-${String(m2+1).padStart(2,'0')}-${String(day).padStart(2,'0')}`; other=true }

    else if(i>=dow+dim){ day=i-dow-dim+1; const m2=calMonth===11?0:calMonth+1,y2=calMonth===11?calYear+1:calYear; ds=`${y2}-${String(m2+1).padStart(2,'0')}-${String(day).padStart(2,'0')}`; other=true }

    else{ day=i-dow+1; ds=`${calYear}-${String(calMonth+1).padStart(2,'0')}-${String(day).padStart(2,'0')}` }

    const dayEvs=allEvents.filter(e=>e.date===ds);

    const isT=ds===today;

    const numEl=isT?`<div class="cdn tdn">${day}</div>`:`<div class="cdn">${day}</div>`;

    cells+=`<div class="cd${other?' om':''}${isT?' td':''}" data-date="${ds}">${numEl}${dayEvs.map(ev=>`<div class="ce ${ev.type==='game'?'gm':'tr'}" data-eid="${ev.id}">${ev.type==='game'?'⚽':'🏃'} ${ev.type==='game'&&ev.opponent?'vs '+ev.opponent:'Treino'}</div>`).join('')}</div>`;

  }

  document.getElementById('cal-body').innerHTML=cells;

  document.getElementById('cal-body').querySelectorAll('.cd').forEach(c=>{

    c.addEventListener('click',()=>openEventModal(null,c.dataset.date));

    c.querySelectorAll('.ce').forEach(ev=>ev.addEventListener('click',e=>{e.stopPropagation();showEventDetail(ev.dataset.eid)}));

  });

  const monthEvs=allEvents.filter(e=>{ const d=new Date(e.date+'T00:00:00'); return d.getFullYear()===calYear&&d.getMonth()===calMonth }).sort((a,b)=>a.date.localeCompare(b.date));

  const lEl=document.getElementById('cal-list');

  lEl.innerHTML=monthEvs.length===0?'<p style="color:var(--muted);font-size:13px">Sem eventos neste mês.</p>':

    monthEvs.map(ev=>`<div class="ei"><div class="etd ${ev.type==='game'?'gm':'tr'}"></div><div class="einfo"><div class="etitle">${ev.type==='game'?`⚽ Jogo${ev.opponent?' vs '+ev.opponent:''}`:'🏃 Treino'}</div><div class="emeta">${fmtDate(ev.date)}${ev.time?' · '+ev.time:''}${ev.location?' · '+ev.location:''}</div>${ev.notes?`<div class="emeta">${ev.notes}</div>`:''}</div><span class="esb ${ev.squad}">${SQUADS[ev.squad]}</span><button class="btn btn-sm btn-g" data-del-event="${ev.id}" style="flex-shrink:0">🗑️</button></div>`).join('');

  lEl.querySelectorAll('[data-del-event]').forEach(b=>b.addEventListener('click',()=>deleteEvent(b.dataset.delEvent)));

}
